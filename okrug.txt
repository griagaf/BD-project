  LOCATION_ID SERIAL PRIMARY KEY,
  LOCATION_NAME VARCHAR(150) NOT NULL,
  CITY VARCHAR(100) NOT NULL,
  ADDRESS TEXT
);

CREATE TABLE MILITARY_UNITS (
  UNIT_ID SERIAL PRIMARY KEY,
  UNIT_NAME VARCHAR(200) NOT NULL,
  UNIT_SHORT_NAME VARCHAR(50),
  UNIT_TYPE VARCHAR(50) CHECK (UNIT_TYPE IN ('Армия', 'Корпус', 'Дивизия', 'Бригада', 'Полк', 'Часть')),
  LOCATION_ID INTEGER REFERENCES LOCATIONS(LOCATION_ID) ON DELETE SET NULL,
  PARENT_UNIT_ID INTEGER REFERENCES MILITARY_UNITS(UNIT_ID) ON DELETE SET NULL,
  COMMANDER_ID INTEGER,
  FORMATION_DATE DATE CHECK (FORMATION_DATE <= CURRENT_DATE),
  STATUS VARCHAR(30) DEFAULT 'Активна' CHECK (STATUS IN ('Активна', 'Расформирована', 'Переформировывается'))
);

CREATE TABLE SUBDIVISIONS (
  SUBDIVISION_ID SERIAL PRIMARY KEY,
  SUBDIVISION_NAME VARCHAR(150) NOT NULL,
  SUBDIVISION_TYPE VARCHAR(30) CHECK (SUBDIVISION_TYPE IN ('Рота', 'Взвод', 'Отделение', 'Батальон', 'Дивизион')),
  COMMANDER_ID INTEGER,
  UNIT_ID INTEGER NOT NULL REFERENCES MILITARY_UNITS(UNIT_ID) ON DELETE CASCADE,
  PARENT_SUBDIVISION_ID INTEGER REFERENCES SUBDIVISIONS(SUBDIVISION_ID) ON DELETE SET NULL,
  STAFF_COUNT INTEGER CHECK (STAFF_COUNT > 0),
  CURRENT_COUNT INTEGER CHECK (CURRENT_COUNT >= 0),
  SPECIALIZATION VARCHAR(200),
  CHECK (CURRENT_COUNT <= STAFF_COUNT)
);

CREATE TABLE PERSONNEL (
  PERSONNEL_ID SERIAL PRIMARY KEY,
  LAST_NAME VARCHAR(100) NOT NULL,
  FIRST_NAME VARCHAR(100) NOT NULL,
  MIDDLE_NAME VARCHAR(100),
  PERSONAL_NUMBER VARCHAR(20) UNIQUE NOT NULL,
  DATE_OF_BIRTH DATE NOT NULL CHECK (DATE_OF_BIRTH <= CURRENT_DATE - INTERVAL '17 years'),
  GENDER CHAR(1) CHECK (GENDER IN ('М', 'Ж')),
  TELEPHONE_NUMBER VARCHAR(15),
  ADDRESS TEXT,
  STATUS VARCHAR(30) NOT NULL CHECK (STATUS IN ('На службе', 'В командировке', 'На лечении', 'В отпуске', 'В запасе')),
  SERVICE_START_DATE DATE NOT NULL CHECK (SERVICE_START_DATE <= CURRENT_DATE),
  SERVICE_END_DATE DATE,
  UNIT_ID INTEGER REFERENCES MILITARY_UNITS(UNIT_ID) ON DELETE SET NULL,
  SUBDIVISION_ID INTEGER REFERENCES SUBDIVISIONS(SUBDIVISION_ID) ON DELETE SET NULL,
  CHECK (SERVICE_END_DATE IS NULL OR SERVICE_END_DATE >= SERVICE_START_DATE)
);

CREATE TABLE BUILDING_TYPES (
  BUILDING_TYPE_ID SERIAL PRIMARY KEY,
  TYPE_NAME VARCHAR(100) NOT NULL,
  TYPICAL_CAPACITY INTEGER CHECK (TYPICAL_CAPACITY > 0),
  DESCRIPTION TEXT
);

CREATE TABLE BUILDINGS (
  BUILDING_ID SERIAL PRIMARY KEY,
  BUILDING_NAME VARCHAR(100) NOT NULL,
  BUILDING_TYPE_ID INTEGER NOT NULL REFERENCES BUILDING_TYPES(BUILDING_TYPE_ID) ON DELETE CASCADE,
  UNIT_ID INTEGER NOT NULL REFERENCES MILITARY_UNITS(UNIT_ID) ON DELETE CASCADE,
  LOCATION_ID INTEGER NOT NULL REFERENCES LOCATIONS(LOCATION_ID) ON DELETE CASCADE,
  CONSTRUCTION_YEAR INTEGER CHECK (CONSTRUCTION_YEAR BETWEEN 1700 AND EXTRACT(YEAR FROM CURRENT_DATE)),
  FLOORS INTEGER CHECK (FLOORS > 0),
  TOTAL_AREA DECIMAL(10,2) CHECK (TOTAL_AREA > 0),
  CAPACITY INTEGER CHECK (CAPACITY > 0),
  CONDITION VARCHAR(50) DEFAULT 'Хорошее' CHECK (CONDITION IN ('Отличное', 'Хорошее', 'Удовлетворительное', 'Требует ремонта', 'Аварийное')),
  LAST_RENOVATION DATE CHECK (LAST_RENOVATION <= CURRENT_DATE),
  RESPONSIBLE_OFFICER_ID INTEGER REFERENCES PERSONNEL(PERSONNEL_ID) ON DELETE SET NULL
);

CREATE TABLE SUBDIVISION_PLACEMENT (
  PLACEMENT_ID SERIAL PRIMARY KEY,
  SUBDIVISION_ID INTEGER NOT NULL REFERENCES SUBDIVISIONS(SUBDIVISION_ID) ON DELETE CASCADE,
  BUILDING_ID INTEGER NOT NULL REFERENCES BUILDINGS(BUILDING_ID) ON DELETE CASCADE,
  START_DATE DATE NOT NULL CHECK (START_DATE <= CURRENT_DATE),
  END_DATE DATE,
  IS_CURRENT BOOLEAN DEFAULT TRUE,
  REASON VARCHAR(200),
  AUTHORIZED_BY INTEGER REFERENCES PERSONNEL(PERSONNEL_ID) ON DELETE SET NULL,
  CHECK (END_DATE IS NULL OR END_DATE >= START_DATE),
  CHECK (IS_CURRENT = TRUE OR END_DATE IS NOT NULL)
);
